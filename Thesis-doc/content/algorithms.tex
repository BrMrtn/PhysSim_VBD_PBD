\chapter{The two algorithms}
\label{chap:Algorithms}
%----------------------------------------------------------------------------

This chapter states XPBD and VBD in the form actually implemented in
\autoref{chap:Methodology}. Both target the same variational problem \eqref{eq:G},
but they reach it through opposite descent directions.

%----------------------------------------------------------------------------
\section{XPBD: dual-variable position projection}
\label{sec:XPBDAlgo}
%----------------------------------------------------------------------------

\subsection{Derivation}
\label{ssec:XPBDDeriv}
For each elastic spring, XPBD associates a constraint $C(\mathbf{x}) = 0$ at zero
deformation with a compliance $\alpha = 1/k$. The quantity
$\lambda = -\alpha^{-1} C$ is the Lagrange multiplier for that constraint and can
be interpreted as the magnitude of the spring force. The augmented Lagrangian of
\eqref{eq:G} with a single substep $h$ has the stationarity conditions
\begin{align}
    \mathbf{M}(\mathbf{x} - \mathbf{y}) - \nabla C(\mathbf{x})^{T}\,\boldsymbol{\lambda} &= \mathbf{0}, \\
    C(\mathbf{x}) + \tilde{\alpha}\,\boldsymbol{\lambda} &= \mathbf{0},
\end{align}
where $\tilde{\alpha} = \alpha / h^{2}$ is the time-step-scaled compliance. XPBD
solves this coupled system with a Gauss--Seidel sweep over the constraints: for
each constraint $j$, the multiplier increment is
\begin{equation}
    \label{eq:XPBDLambda}
    \Delta \lambda_{j} \;=\; \frac{-C_{j}(\mathbf{x}) - \tilde{\alpha}_{j}\, \lambda_{j}}
                                {\nabla C_{j}^{\!\top}\,\mathbf{M}^{-1}\nabla C_{j}
                                 + \tilde{\alpha}_{j}}.
\end{equation}
XPBD then omits the Hessian by approximating the stiffness matrix with the mass
matrix, which gives the position correction
$\Delta \mathbf{x} = \mathbf{M}^{-1}\nabla C_{j}^{\!\top}\,\Delta \lambda_{j}$.
Velocities are recovered at the end of each substep by
$\mathbf{v}_{t+h} = (\mathbf{x}_{t+h} - \mathbf{x}_{t}) / h$.

\subsection{Algorithm}
\label{ssec:XPBDPseudo}
\autoref{alg:XPBDCore} shows the simplified form of XPBD used for the core
comparison: no damping or collision handling, and only one type of constraint.
The full implemented version --- with damping, self-collision, and more detailed
pseudocode --- is given in \autoref{sec:AppXPBD}.

\begin{algorithm}[!ht]
\caption{XPBD step (core comparison). $n_{\mathrm{sub}}$ substeps, each with
         $n_{\mathrm{iter}}$ Gauss--Seidel iterations over the constraints.}
\label{alg:XPBDCore}
\begin{algorithmic}[1]
\State $h \gets dt / n_{\mathrm{sub}}$ \Comment{substep size}
\For{$s = 1$ \textbf{to} $n_{\mathrm{sub}}$}
    \ForAll{vertices $i$}
        \State $\mathbf{v}_i \gets \mathbf{v}_i + h\,w_i\,\mathbf{f}_{\mathrm{ext}}(\mathbf{x}_i)$
               \Comment{$w_i = 1/m_i$; gravity folded in}
        \State $\mathbf{p}_i \gets \mathbf{x}_i + h\,\mathbf{v}_i$
               \Comment{inertial prediction $\mathbf{y}_i$}
    \EndFor
    \ForAll{constraints $j$}\ $\lambda_j \gets 0$ \EndFor
        \Comment{multipliers reset every substep}
    \For{$\mathrm{it} = 1$ \textbf{to} $n_{\mathrm{iter}}$}
        \State \Call{ProjectConstraints}{$C_1,\dots,C_M,\ \mathbf{p},\ \boldsymbol{\lambda}$}
    \EndFor
    \ForAll{vertices $i$}
        \State $\mathbf{v}_i \gets (\mathbf{p}_i - \mathbf{x}_i)/h$;\quad $\mathbf{x}_i \gets \mathbf{p}_i$
    \EndFor
\EndFor
\Statex
\Procedure{ProjectConstraints}{$C_1,\dots,C_M,\ \mathbf{p},\ \boldsymbol{\lambda}$}
    \ForAll{constraints $j$} \Comment{one Gauss--Seidel sweep}
        \State $\tilde{\alpha}_j \gets \alpha_j / h^2$
        \State $\Delta\lambda_j \gets \dfrac{-C_j(\mathbf{p}) - \tilde{\alpha}_j\,\lambda_j}
                {\nabla C_j\,\mathbf{M}^{-1}\nabla C_j^{\!\top} + \tilde{\alpha}_j}$
        \State $\lambda_j \gets \lambda_j + \Delta\lambda_j$
        \State $\mathbf{p} \gets \mathbf{p} + \mathbf{M}^{-1}\nabla C_j^{\!\top}\,\Delta\lambda_j$
    \EndFor
\EndProcedure
\end{algorithmic}
\end{algorithm}

\subsection{Characteristics}
\label{ssec:XPBDChar}
XPBD has four defining characteristics. It is \emph{dual-variable}: an iteration's
state is encoded in the multipliers $\boldsymbol{\lambda}$, and positions are
derived from them. It handles \emph{stiffness through compliance}, since material
stiffness enters only via $\tilde{\alpha} = 1/(k h^{2})$, so very stiff
constraints naturally degenerate to hard projection ($\tilde\alpha \to 0$). It
uses \emph{first-order gradient information only}: each constraint contributes its
gradient $\nabla C_j$, and no Hessian of $E$ is built. Finally, it \emph{prefers
substeps over iterations}: following the small-steps result, I use many short
substeps with a single inner iteration.

\section{VBD: primal block-coordinate descent}
\label{sec:VBDAlgo}
%----------------------------------------------------------------------------

\subsection{Derivation}
\label{ssec:VBDDeriv}
VBD performs block-coordinate descent on $G$ in \eqref{eq:G}, with one block per
vertex. Holding all other vertices fixed, the update for vertex $i$ is the Newton
step on the local 3-vector problem
\begin{equation}
    \mathbf{x}_{i} \;\leftarrow\; \mathbf{x}_{i} - \mathbf{H}_{i}^{-1}\,\mathbf{g}_{i},
\end{equation}
where
\begin{align}
    \mathbf{g}_{i} &= \frac{m_{i}}{h^{2}}\,(\mathbf{x}_{i} - \mathbf{y}_{i})
                      + \sum_{e \in \mathcal{N}_i} \nabla_{i} E_{e}(\mathbf{x}), \\
    \mathbf{H}_{i} &= \frac{m_{i}}{h^{2}}\,\mathbf{I}_{3}
                      + \sum_{e \in \mathcal{N}_i} \nabla_{ii}^{2}\, E_{e}(\mathbf{x}).
\end{align}
Here $\mathbf{g}_{i}$ is the gradient of the local variational energy around
vertex $i$, $\mathcal{N}_i$ is the set of force elements acting on vertex $i$, and
$\mathbf{H}_{i}$ is the local Hessian; $\nabla_{ii}^{2}\, E_{e}(\mathbf{x})$ is the
second derivative of the energy of element $e$ with respect to vertex $i$'s
position. For a distance-spring energy
$E_{e} = \tfrac{1}{2} k\,(\|\mathbf{x}_{i} - \mathbf{x}_{j}\| - \ell_{0})^{2}$, the
per-vertex gradient and Hessian are closed-form and inexpensive to evaluate.

\subsection{Algorithm}
\label{ssec:VBDPseudo}
As with XPBD, the form below is the simplified, core-comparison version: no
damping and no collisions. Here I use the fixed-iteration-count variant in place
of the convergence-based early-stopping one. The full implemented version --- with
adaptive initialization, damping, and self-collision --- is given in
\autoref{sec:AppVBD}.

\begin{algorithm}[!ht]
\caption{VBD step (core comparison). $n_{\mathrm{sub}}$ substeps, each with
         $n_{\mathrm{iter}}$ block-descent iterations over the vertices.}
\label{alg:VBDCore}
\begin{algorithmic}[1]
\State $h \gets dt / n_{\mathrm{sub}}$
\For{$s = 1$ \textbf{to} $n_{\mathrm{sub}}$}
    \ForAll{vertices $i$}
        \State $\mathbf{y}_i \gets \mathbf{x}_i + h\,\mathbf{v}_i + h^2 w_i\,\mathbf{f}_{\mathrm{ext}}(\mathbf{x}_i)$
               \Comment{inertial target}
        \State $\mathbf{x}^{\mathrm{prev}}_i \gets \mathbf{x}_i$;\quad $\mathbf{x}_i \gets \mathbf{y}_i$
    \EndFor
    \For{$\mathrm{it} = 1$ \textbf{to} $n_{\mathrm{iter}}$}
        \State \Call{Sweep}{$\mathbf{x},\ \mathbf{y}$}
               \Comment{one Gauss--Seidel pass over vertices}
    \EndFor
    \ForAll{vertices $i$}
        \State $\mathbf{v}_i \gets (\mathbf{x}_i - \mathbf{x}^{\mathrm{prev}}_i)/h$
    \EndFor
\EndFor
\Statex
\Procedure{Sweep}{$\mathbf{x},\ \mathbf{y}$}
    \ForAll{vertices $i$}
        \State $\mathbf{g}_i \gets \dfrac{m_i}{h^2}(\mathbf{x}_i - \mathbf{y}_i)
                + \displaystyle\sum_{e \in \mathcal{N}(i)} \nabla_i E_e(\mathbf{x})$
               \Comment{negative force}
        \State $\mathbf{H}_i \gets \dfrac{m_i}{h^2}\mathbf{I}_3
                + \displaystyle\sum_{e \in \mathcal{N}(i)} \nabla^2_{ii} E_e(\mathbf{x})$
        \State $\mathbf{x}_i \gets \mathbf{x}_i - \mathbf{H}_i^{-1}\mathbf{g}_i$
               \Comment{local $3\times 3$ Newton step}
    \EndFor
\EndProcedure
\end{algorithmic}
\end{algorithm}

\subsection{Characteristics}
\label{ssec:VBDChar}
VBD likewise has four defining characteristics. It is \emph{primal}: an
iteration's state is the positions $\mathbf{x}$, with no extra multipliers. It is
\emph{second-order locally}, since each block update is a Newton step on the true
elastic Hessian restricted to one vertex. It \emph{prefers iterations over
substeps}: VBD is normally used with one long step subdivided into many block
iterations, and the implicit Hessian handles stiffness even at larger $h$.
Finally, it is \emph{parallel-friendly}: vertices that share no constraint can be
updated independently through graph coloring.

\section{Side-by-side: where the algorithms differ}
\label{sec:Sidebyside}
%----------------------------------------------------------------------------

\begin{table}[!ht]
    \footnotesize
    \centering
    \begin{tabular}{l l l}
        \toprule
                                  & XPBD                          & VBD \\
        \midrule
        Variable updated          & multipliers $\boldsymbol{\lambda}$ & positions $\mathbf{x}$ \\
        Order of information      & gradient $\nabla C$           & gradient + Hessian of $E$ \\
        Block size                & per-constraint                & per-vertex \\
        Stiffness handled by      & compliance $\tilde{\alpha}$   & implicit Hessian \\
        Typical configuration     & many substeps, $1$ iteration  & one step, many iterations \\
        Parallelism strategy      & constraint coloring (harder)  & vertex coloring (easier) \\
        \bottomrule
    \end{tabular}
    \caption{Algorithmic contrast between XPBD and VBD.}
    \label{tab:XPBDvsVBD}
\end{table}

The first and third rows capture the distinction that the primal/dual lens
(\autoref{sec:PrimalDualLens}) builds on, and the one the experiments return to
when interpreting each result.
