%----------------------------------------------------------------------------
\section{XPBD: dual-variable position projection}
\label{sec:XPBDAlgo}
%----------------------------------------------------------------------------

\subsection{Derivation}
\label{ssec:XPBDDeriv}
\begin{itemize}
    \item For each elastic spring, associate a constraint $C(\mathbf{x}) = 0$ at zero deformation and a compliance $\alpha = 1/k$.
    \item $\lambda = -\alpha^{-1} C$ is the Lagrange multiplier for the constraint, which can be interpreted as the spring force magnitude.
    \item The augmented Lagrangian of \eqref{eq:G} with a single substep $h$ has stationarity conditions, and the constrained equation of motion:
\end{itemize}
\begin{align}
    \mathbf{M}(\mathbf{x} - \mathbf{y}) - \nabla C(\mathbf{x})^{T}\,\boldsymbol{\lambda} &= \mathbf{0}, \\
    C(\mathbf{x}) + \tilde{\alpha}\,\boldsymbol{\lambda} &= \mathbf{0},
\end{align}
\begin{itemize}
    \item $\tilde{\alpha} = \alpha / h^{2}$ is the time-step-scaled compliance.
    \item XPBD solves this coupled system by a Gauss--Seidel sweep over constraints: for each constraint $j$, the multiplier increment is
\end{itemize}
\begin{equation}
    \label{eq:XPBDLambda}
    \Delta \lambda_{j} \;=\; \frac{-C_{j}(\mathbf{x}) - \tilde{\alpha}_{j}\, \lambda_{j}}
                                {\nabla C_{j}^{\!\top}\,\mathbf{M}^{-1}\nabla C_{j}
                                 + \tilde{\alpha}_{j}},
\end{equation}
\begin{itemize}
    \item After XPBD omits the calculation of the Hessian by approximating their stiffness matrix with the mass matrix, we get the following equation for the position correction:
    \item $\Delta \mathbf{x} = \mathbf{M}^{-1}\nabla C_{j}^{\!\top}\,\Delta \lambda_{j}$.
    \item Velocities are recovered at substep end by $\mathbf{v}_{t+h} = (\mathbf{x}_{t+h} - \mathbf{x}_{t}) / h$.
\end{itemize}

\subsection{Algorithm}
\label{ssec:XPBDPseudo}
The simplified form of the XPBD algorithm used for the core comparisons can be seen on \autoref{alg:XPBDCore}: no damping or
collision handling, only one type of constraint. The full implemented version, with
damping and self-collision and with more detailed pseudocode is in \autoref{sec:AppXPBD}.

\begin{algorithm}[!ht]
\caption{XPBD step (core comparison). $n_{\mathrm{sub}}$ substeps, each with
         $n_{\mathrm{iter}}$ Gauss--Seidel iterations over the constraints.}
\label{alg:XPBDCore}
\begin{algorithmic}[1]
\State $h \gets dt / n_{\mathrm{sub}}$ \Comment{substep size}
\For{$s = 1$ \textbf{to} $n_{\mathrm{sub}}$}
    \ForAll{vertices $i$}
        \State $\mathbf{v}_i \gets \mathbf{v}_i + h\,w_i\,\mathbf{f}_{\mathrm{ext}}(\mathbf{x}_i)$
               \Comment{$w_i = 1/m_i$; gravity folded in}
        \State $\mathbf{p}_i \gets \mathbf{x}_i + h\,\mathbf{v}_i$
               \Comment{inertial prediction $\mathbf{y}_i$}
    \EndFor
    \ForAll{constraints $j$}\ $\lambda_j \gets 0$ \EndFor
        \Comment{multipliers reset every substep}
    \For{$\mathrm{it} = 1$ \textbf{to} $n_{\mathrm{iter}}$}
        \State \Call{ProjectConstraints}{$C_1,\dots,C_M,\ \mathbf{p},\ \boldsymbol{\lambda}$}
    \EndFor
    \ForAll{vertices $i$}
        \State $\mathbf{v}_i \gets (\mathbf{p}_i - \mathbf{x}_i)/h$;\quad $\mathbf{x}_i \gets \mathbf{p}_i$
    \EndFor
\EndFor
\Statex
\Procedure{ProjectConstraints}{$C_1,\dots,C_M,\ \mathbf{p},\ \boldsymbol{\lambda}$}
    \ForAll{constraints $j$} \Comment{one Gauss--Seidel sweep}
        \State $\tilde{\alpha}_j \gets \alpha_j / h^2$
        \State $\Delta\lambda_j \gets \dfrac{-C_j(\mathbf{p}) - \tilde{\alpha}_j\,\lambda_j}
                {\nabla C_j\,\mathbf{M}^{-1}\nabla C_j^{\!\top} + \tilde{\alpha}_j}$
        \State $\lambda_j \gets \lambda_j + \Delta\lambda_j$
        \State $\mathbf{p} \gets \mathbf{p} + \mathbf{M}^{-1}\nabla C_j^{\!\top}\,\Delta\lambda_j$
    \EndFor
\EndProcedure
\end{algorithmic}
\end{algorithm}

\subsection{Characteristics}
\label{ssec:XPBDChar}
\begin{itemize}
    \item \emph{Dual-variable.} An iteration's state is encoded in the multipliers $\boldsymbol{\lambda}$; positions are derived.
    \item \emph{Stiffness-by-compliance.} Material stiffness enters only through $\tilde{\alpha} = 1/(k h^{2})$, so very stiff constraints naturally degenerate to projection ($\tilde\alpha \to 0$).
    \item \emph{First-order gradient information only.} Each constraint contributes its gradient $\nabla C_j$; no Hessian of $E$ is built.
    \item \emph{Substeps preferred over iterations.} Following the small-steps result, I use many short substeps with one inner iteration.
\end{itemize}

\section{VBD: primal block-coordinate descent}
\label{sec:VBDAlgo}
%----------------------------------------------------------------------------

\subsection{Derivation}
\label{ssec:VBDDeriv}
\begin{itemize}
    \item VBD performs block-coordinate descent on $G$ in \eqref{eq:G} with one block per vertex.
    \item Holding all other vertices fixed, the update for vertex $i$ is the Newton step on the local 3-vector problem:
\end{itemize}
\begin{equation}
    \mathbf{x}_{i} \;\leftarrow\; \mathbf{x}_{i} - \mathbf{H}_{i}^{-1}\,\mathbf{g}_{i},
\end{equation}
where
\begin{align}
    \mathbf{g}_{i} &= \frac{m_{i}}{h^{2}}\,(\mathbf{x}_{i} - \mathbf{y}_{i})
                      + \sum_{e \in \mathcal{N_i}} \nabla_{i} E_{e}(\mathbf{x}), \\
    \mathbf{H}_{i} &= \frac{m_{i}}{h^{2}}\,\mathbf{I}_{3}
                      + \sum_{e \in \mathcal{N_i}} \nabla_{ii}^{2}\, E_{e}(\mathbf{x}).
\end{align}
\begin{itemize}
    \item where $\mathbf{g}_{i}$ is the derivative of the local variational energy around vertex $i$, \mathcal{N_i} is the set of force elements acting on vertex $i$, and $\mathbf{H}_{i}$ is the local Hessian.
    \item $\nabla_{ii}^{2}\, E_{e}(\mathbf{x})$ is second derivative of the energy of element $e$ with respect to vertex $i$'s position.
    \item For a distance-spring energy $E_{e} = \tfrac{1}{2} k\,(\|\mathbf{x}_{i} - \mathbf{x}_{j}\| - \ell_{0})^{2}$, the per-vertex gradient and Hessian are closed-form and inexpensive.
\end{itemize}

\subsection{Algorithm}
\label{ssec:VBDPseudo}
As with XPBD, the form below is the simplified variant of the core-comparison version: no damping, no
collisions. Here I use the fixed iteration count version in place of the convergence early-stopping one.
The full implemented version, with adaptive initialisation, damping and self-collision,
is in \autoref{sec:AppVBD}.

\begin{algorithm}[!ht]
\caption{VBD step (core comparison). $n_{\mathrm{sub}}$ substeps, each with
         $n_{\mathrm{iter}}$ block-descent iterations over the vertices.}
\label{alg:VBDCore}
\begin{algorithmic}[1]
\State $h \gets dt / n_{\mathrm{sub}}$
\For{$s = 1$ \textbf{to} $n_{\mathrm{sub}}$}
    \ForAll{vertices $i$}
        \State $\mathbf{y}_i \gets \mathbf{x}_i + h\,\mathbf{v}_i + h^2 w_i\,\mathbf{f}_{\mathrm{ext}}(\mathbf{x}_i)$
               \Comment{inertial target}
        \State $\mathbf{x}^{\mathrm{prev}}_i \gets \mathbf{x}_i$;\quad $\mathbf{x}_i \gets \mathbf{y}_i$
    \EndFor
    \For{$\mathrm{it} = 1$ \textbf{to} $n_{\mathrm{iter}}$}
        \State \Call{Sweep}{$\mathbf{x},\ \mathbf{y}$}
               \Comment{one Gauss--Seidel pass over vertices}
    \EndFor
    \ForAll{vertices $i$}
        \State $\mathbf{v}_i \gets (\mathbf{x}_i - \mathbf{x}^{\mathrm{prev}}_i)/h$
    \EndFor
\EndFor
\Statex
\Procedure{Sweep}{$\mathbf{x},\ \mathbf{y}$}
    \ForAll{vertices $i$}
        \State $\mathbf{g}_i \gets \dfrac{m_i}{h^2}(\mathbf{x}_i - \mathbf{y}_i)
                + \displaystyle\sum_{e \in \mathcal{N}(i)} \nabla_i E_e(\mathbf{x})$
               \Comment{negative force}
        \State $\mathbf{H}_i \gets \dfrac{m_i}{h^2}\mathbf{I}_3
                + \displaystyle\sum_{e \in \mathcal{N}(i)} \nabla^2_{ii} E_e(\mathbf{x})$
        \State $\mathbf{x}_i \gets \mathbf{x}_i - \mathbf{H}_i^{-1}\mathbf{g}_i$
               \Comment{local $3\times 3$ Newton step}
    \EndFor
\EndProcedure
\end{algorithmic}
\end{algorithm}

\subsection{Characteristics}
\label{ssec:VBDChar}
\begin{itemize}
    \item \emph{Primal.} An iteration's state is positions $\mathbf{x}$; no extra multipliers.
    \item \emph{Second-order locally.} Each block update is a Newton step on the true elastic Hessian restricted to one vertex.
    \item \emph{Iterations over substeps.} VBD is normally used with one long step subdivided by many block iterations; the implicit Hessian handles stiffness with larger $h$ too.
    \item \emph{Parallel-friendly.} Vertices sharing no constraint can be updated independently (graph colouring).
\end{itemize}
